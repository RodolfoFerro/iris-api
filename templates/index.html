Iris Classifier API

Try it out: 

import requests

# Set metadata:
headers = {'Content-type': 'application/json'}
input_values = {
    # Edit and try with your own data:
    'sepal_length': 7.7,
    'sepal_width': 2.8,
    'petal_length': 6.7,
    'petal_width': 2.0
}

# Get response:
response = requests.post('http://127.0.0.1:5000/api/v0.0/predict',
                         headers=headers, 
                         json=input_values)

# If response's status is 200:
if response.status_code == requests.codes.ok:
      # Pretty print response:
      pprint(response.json())
